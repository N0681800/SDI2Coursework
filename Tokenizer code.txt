
#include <iostream>
#include <string>
using namespace std;


int main()
{
	char Test[] = "Hello///My///Name///Is///Josh";
	char *context = NULL;
	char *token = strtok_s(Test, "///",&context);
	
	while (token != NULL)
	{
		cout<<"\n"<<token;
		token = strtok_s(NULL, "///",&context);
	}

	getchar();
	return 0;
}
